<!-- QUESTION: qtype=fill_blank tags=热力学,等压过程,热量,定压摩尔热容 difficulty=3 chapter=第四章 热力学定律 qid=Q0642 -->

<!-- ANSWER -->

<!-- EXPLANATION -->
